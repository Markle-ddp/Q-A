
## 切片问题
```go
package main

import (
	"encoding/json"
	"fmt"
)

type AutoGenerated struct {
	Age   int    `json:"age"`
	Name  string `json:"name"`
	Child []int  `json:"child"`
}

func main() {

	var aa []int
	a := AutoGenerated{}

	jsonStr2 := `{"age": 13,"name": "potter2", "child":[3,4,5,6,7]}`
	json.Unmarshal([]byte(jsonStr2), &a)

	aa = a.Child
	println(aa)     // output: [5/6]0xc000014150
	fmt.Println(aa) // output: [3 4 5 6 7]

	jsonStr3 := `{"age": 124,"name": "potter3", "child":[8,9]}`
	json.Unmarshal([]byte(jsonStr3), &a)
	fmt.Println(aa) // output: [8 9 5 6 7]
}

```

> 切片指向的底层数组数据发生变更导致
